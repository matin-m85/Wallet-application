//////////////////////////////💰 Wallet API Tests//////////////////////////////


# 🪙 Create a wallet for the user

# 💵 Add balance to the wallet
curl -X POST http://localhost:8082/api/wallet/redeem \
     -H "Content-Type: application/json" \
     -d '{
       "phone": "09120000000",
       "amount": 50000,
       "reference": "initial-topup"
     }'


# 📊 Get wallet information
curl http://localhost:8082/api/wallet/balance/09120000000


# 📜 List all redeemed wallets
curl http://localhost:8082/api/wallet/redeemed
